### 7.2 circos.conf file
Here is an example of a circos.conf file, you can modify all the parameters

```
<<include etc/housekeeping.conf>>
<<include etc/colors_fonts_patterns.conf>>
<<include ideogram.conf>>
<<include ticks.conf>>

# Karyotype configuration
karyotype = data/karyotype.txt

# Bloque de imagen
<image>
<<include etc/image.conf>>
</image>

# Chromosomes and units definitions
chromosomes_units = 1000
chromosomes = contig1
chromosomes_display_default = yes




# Fonts configuration
<fonts>
    <<include etc/fonts.conf>>
</fonts>



# Max limits
max_ticks*            = 5000
max_ideograms*        = 200
max_links*            = 25000
max_points_per_track* = 30000

# Specific data
<<include conf/data.conf>>
#<<include conf/links.conf>>

# Colors
<colors>
dblue      = 0,0,255
green      = 0,255,0
red        = 255,0,0
white      = 255,255,255
chromosome = dblue
plasmid    = green
gene       = red
</colors>
```
